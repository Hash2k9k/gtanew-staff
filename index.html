}); } // Fonction pour mettre à jour le nombre de tickets function updateTickets(input) { const index = input.getAttribute('data-index'); const newValue = parseInt(input.value, 10); const staffData = JSON.parse(localStorage.getItem('staff')) || []; staffData[index].tickets = newValue; localStorage.setItem('staff', JSON.stringify(staffData)); } // Fonction pour supprimer un membre du staff function deleteStaff(button) { const index = button.getAttribute('data-index'); const staffData = JSON.parse(localStorage.getItem('staff')) || []; staffData.push(newStaff); localStorage.setItem('staff', JSON.stringify(staffData)); // Réinitialise le formulaire et recharge les données nameInput.value = ''; ticketsInput.value = '0'; loadStaffData(); } // Événement pour ajouter un membre document.getElementById('add-staff-form').addEventListener('submit', addStaff); // Charger les données du staff lorsque la page est chargée loadStaffData();
